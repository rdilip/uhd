115  // The operator is used when putting the severity level to log
116  template <typename CharT, typename TraitsT>
117  inline std::basic_ostream<CharT, TraitsT>&
118  operator<<(std::basic_ostream<CharT, TraitsT>& strm, severity_level level)
119  {
120  static const char* strings[] = {"TRACE", "DEBUG", "INFO",
121  "WARNING", "ERROR", "FATAL"
122  };
123 
124  if (static_cast<std::size_t>(level) < sizeof(strings) / sizeof(*strings)) {
125  strm << strings[level];
126  } else {
127  strm << static_cast<int>(level);
128  }
129 
130  return strm;
131  }
